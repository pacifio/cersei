export function Tag({ type }) {
  const styles = {
    added: 'bg-fd-accent text-fd-accent-foreground border-fd-border',
    changed: 'bg-fd-secondary text-fd-secondary-foreground border-fd-border',
    fixed: 'bg-fd-muted text-fd-muted-foreground border-fd-border',

  return (
    <span className={`inline-flex items-center text-[10px] uppercase px-1.5 py-0.5 rounded border align-middle mr-1 ${styles[type] || styles.added}`}>
      {type}
    </span>
  );
}

<Tag type="changed" /> **`abstract login <provider>`** accepts any provider registered in the `cersei-provider` registry (Google, Groq, DeepSeek, xAI, Mistral, Together, Fireworks, Perplexity, Cerebras, OpenRouter, Cohere, SambaNova) â€” previously only `anthropic` and `openai` were wired up. <Tag type="fixed" /> **Auto-default silently picking Ollama.** Two changes: (1) `registry::available()` TCP-probes local providers (`env_keys` empty) with a 200ms check on their `api_base`, so `abstract login status` distinguishes `available (local)` from `not running`. <Tag type="fixed" /> **`abstract login google`** (and every other registered provider) no longer rejected with "Unknown provider".
